147Direction move_elevator(Elevator *elevator) { //Decides what direction the motor should move
148 if (elevator->direction != DIRN_STOP) { //Dont change direction if already moving
149 return elevator->direction;
150 }
151 if (elevator->stopFlag) { //Don't move if stop button is down
152 return DIRN_STOP;
153 }
154 int8_t last_dirr = elevator->last_direction; //Decide direction based on last direction
155 switch (last_dirr) {
156 case DIRN_UP:
157 for (int i = elevator->floor; i < 4; i++){
158 if (elevator->queue.orders[i][BUTTON_HALL_UP] || elevator->queue.orders[i][BUTTON_CAB]){ //First check if in same as last direction
159 return DIRN_UP;
160 }
161 }
162 for (int i = 0; i < 4;i++){ //If not check if there are orders
163 for (int j = 0; j < 3; j++){
164 if (elevator->queue.orders[i][j]){
165 if (i < elevator->floor){ //Where is the order relative to the elevator
166 return DIRN_DOWN;
167 }
168 else{
169 return DIRN_UP;
170 }
171 }
172 }
173 }
174 return DIRN_STOP; //If no orders, stop
175 break;
176 case DIRN_DOWN:
177 for (int i = elevator->floor; i >= 0; i--){ //Same as above, but opposite last direction
178 if (elevator->queue.orders[i][BUTTON_HALL_DOWN] || elevator->queue.orders[i][BUTTON_CAB]){
179 return DIRN_DOWN;
180 }
181 }
182 for (int i = 0; i < 4; i++){
183 for (int j = 0; j < 3; j++){
184 if (elevator->queue.orders[i][j]){
185 if (i < elevator->floor){
186 return DIRN_DOWN;
187 }
188 else{
189 return DIRN_UP;
190 }
191 }
192 }

194 return DIRN_STOP;

209bool check_if_stop(Elevator *elevator) {

213 if (elevator->direction == DIRN_UP && elevator->floor == 3) { //Prevent out of bounds top

219 int8_t lastOrder = 0; //To check if there is exactly 1 order left

221 case (DIRN_UP):
222 if (elevator->queue.orders[elevator->floor][BUTTON_HALL_UP] || elevator->queue.orders[elevator->floor][BUTTON_CAB]) {
223 return true;

227 return true;
